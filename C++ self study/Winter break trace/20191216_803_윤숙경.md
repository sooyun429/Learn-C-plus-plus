  ```c++
  // break/continue
  // 중첩for문
  
  // 연산자: + - / * % += -= ++ -- && || !
  // 제어문: if() switch()
  // 반복문: while() for()
  
  #include <stdio.h>
  
  int main() {
  	// 중첩for문
  	int n;
  	scanf("%d", &n);
  	for (int i = 0; i < n; i++) {
  		for (int j = 0; j <= i; j++) {
  			printf("%d ", 2 * j + 1);
  		}
  		printf("\n");
  	}
  }
  ```
